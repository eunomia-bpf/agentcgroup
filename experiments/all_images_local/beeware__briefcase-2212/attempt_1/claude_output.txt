+    def _fix_pth_file(self, support_path):
+        """Fix the Python .pth file to enable site module loading.
+
+        The embeddable Python package contains a pythonXX._pth file with
+        'import site' commented out by default. This prevents .pth files
+        in site-packages from being processed, which breaks packages
+        like pywin32 that use .pth files to add paths.
+
+        :param support_path: The path where the support package was unpacked.
+        """
+        # Find the .pth file in the support package
+        pth_file = support_path / f"python{self.python_version_tag.replace('.', '')}._pth"
+
+        if pth_file.exists():
+            try:
+                # Read the current content
+                content = pth_file.read_text(encoding="utf-8")
+
+                # Uncomment the import site line
+                if "import site" not in content:
+                    # The file doesn't have 'import site' at all, or it's not the standard format
+                    # We'll add site module loading capability
+                    if "pythonXX.zip" in content or f"python{self.python_version_tag}.zip" in content:
+                        # Add site module loading after the python zip line
+                        lines = content.splitlines()
+                        for i, line in enumerate(lines):
+                            if line.strip().startswith("pythonXX.zip") or line.strip().startswith(f"python{self.python_version_tag}.zip"):
+                                # Add import site after this line
+                                lines.insert(i + 1, "import site")
+                                break
+                        content = "\n".join(lines)
+                    else:
+                        # Insert import site after the first line
+                        lines = content.splitlines()
+                        if lines and not lines[0].startswith("#"):
+                            lines.insert(1, "import site")
+                            content = "\n".join(lines)
+
+                # Write the fixed content back
+                pth_file.write_text(content, encoding="utf-8")
+                self.console.verbose(f"Fixed {pth_file.name} to enable site module")
+            except Exception as e:
+                self.console.warning(
+                    f"Failed to fix .pth file: {e}. Packages that use .pth files may not work correctly."
+                )
